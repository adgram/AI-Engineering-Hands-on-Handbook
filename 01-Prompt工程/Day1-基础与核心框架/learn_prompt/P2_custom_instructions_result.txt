<script>
    const { createApp, ref, computed, onMounted, nextTick } = Vue;

    // ===== 应用定义 =====
    createApp({
        setup() {
            // ----- 状态 -----
            const difficulty = ref('easy'); // 'easy' | 'medium' | 'hard'
            const score = ref(0);
            const total = ref(0);
            const userAnswer = ref('');
            const answered = ref(false);
            const feedbackMessage = ref('');
            const feedbackType = ref(''); // '' | 'correct' | 'wrong'
            const questionKey = ref(0);

            // 当前题目
            const question = ref({
                a: 0,
                b: 0,
                operator: '+',
                correctAnswer: 0,
            });

            // DOM 引用
            const inputRef = ref(null);

            // ----- 配置 -----
            const difficulties = [
                { label: '简单 1-10', value: 'easy' },
                { label: '中等 1-50', value: 'medium' },
                { label: '困难 1-100', value: 'hard' },
            ];

            const difficultyRange = computed(() => {
                const map = { easy: 10, medium: 50, hard: 100 };
                return map[difficulty.value] || 10;
            });

            const difficultyLabel = computed(() => {
                const map = { easy: '简单 · 1~10', medium: '中等 · 1~50', hard: '困难 · 1~100' };
                return map[difficulty.value] || '简单 · 1~10';
            });

            // ----- 计算属性 -----
            const accuracy = computed(() => {
                if (total.value === 0) return 0;
                return Math.round((score.value / total.value) * 100);
            });

            const canSubmit = computed(() => {
                return !answered.value && userAnswer.value !== '' && userAnswer.value !== null;
            });

            const showCorrectAnswer = computed(() => {
                return feedbackType.value === 'wrong';
            });

            const feedbackClass = computed(() => {
                if (feedbackType.value === 'correct') return 'correct';
                if (feedbackType.value === 'wrong') return 'wrong';
                return '';
            });

            // ----- 工具函数 -----
            function randomInt(min, max) {
                return Math.floor(Math.random() * (max - min + 1)) + min;
            }

            function generateQuestion() {
                const max = difficultyRange.value;
                const isAddition = Math.random() < 0.5;
                let a, b, operator, correctAnswer;

                if (isAddition) {
                    a = randomInt(1, max);
                    b = randomInt(1, max);
                    operator = '+';
                    correctAnswer = a + b;
                } else {
                    // 减法：确保 a >= b，结果非负
                    a = randomInt(1, max);
                    b = randomInt(1, max);
                    if (a < b)[a, b] = [b, a];
                    operator = '−';
                    correctAnswer = a - b;
                }

                return { a, b, operator, correctAnswer };
            }

            // ----- 核心方法 -----
            function nextQuestion() {
                question.value = generateQuestion();
                questionKey.value++;
                userAnswer.value = '';
                answered.value = false;
                feedbackMessage.value = '';
                feedbackType.value = '';
                // 自动聚焦
                nextTick(() => {
                    inputRef.value?.focus();
                });
            }

            function submitAnswer() {
                if (!canSubmit.value) return;

                const userAns = Number(userAnswer.value);
                if (isNaN(userAns)) return;

                const isCorrect = userAns === question.value.correctAnswer;
                total.value++;

                if (isCorrect) {
                    score.value++;
                    feedbackMessage.value = '✅ 完全正确！';
                    feedbackType.value = 'correct';
                } else {
                    feedbackMessage.value = '❌ 答错了，正确答案是';
                    feedbackType.value = 'wrong';
                }

                answered.value = true;

                // 延迟自动跳转下一题
                setTimeout(() => {
                    nextQuestion();
                }, isCorrect ? 800 : 1600);
            }

            function setDifficulty(val) {
                if (difficulty.value === val) return;
                difficulty.value = val;
                resetGame();
            }

            function resetGame() {
                score.value = 0;
                total.value = 0;
                feedbackMessage.value = '';
                feedbackType.value = '';
                nextQuestion();
            }

            // ----- 生命周期 -----
            onMounted(() => {
                nextQuestion();
                // 全局键盘监听：防止回车导致页面刷新
                document.addEventListener('keydown', (e) => {
                    if (e.key === 'Enter' && e.target?.tagName === 'INPUT') {
                        e.preventDefault();
                    }
                });
            });

            // ----- 暴露给模板 -----
            return {
                difficulty,
                difficulties,
                difficultyLabel,
                score,
                total,
                accuracy,
                userAnswer,
                answered,
                feedbackMessage,
                feedbackType,
                feedbackClass,
                showCorrectAnswer,
                question,
                questionKey,
                canSubmit,
                inputRef,
                submitAnswer,
                setDifficulty,
                resetGame,
            };
        },
    }).mount('#app');
</script>

### 算术练习说明

从生成题目到统计反馈，整个练习流程清晰流畅：

- **题目生成逻辑**：每次随机选择加法或减法，减法保证被减数不小于减数，结果非负。难度分为简单（1-10）、中等（1-50）、困难（1-100），点击底部按钮即可切换，题目会自动重置。
- **交互与反馈**：在输入框中填写答案后，点击“提交”或按 Enter 键即可。系统会立即判断对错，并显示正确答案（答错时）。正确或错误反馈会停留片刻（正确0.8秒，错误1.6秒），然后自动进入下一题，无需手动操作。
- **统计与视觉**：顶部实时更新得分、总题数和正确率。每次切换题目，数字区域有淡入动画，反馈区也有颜色区分（绿色正确 / 红色错误），让学习过程更直观。
