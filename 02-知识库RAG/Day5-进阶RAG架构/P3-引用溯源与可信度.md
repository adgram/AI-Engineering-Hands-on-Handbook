## 为什么要做引用溯源？

在 RAG 系统中标注信息来源，主要解决可信度、可验证、可追溯和合规四方面需求。

1. **可信度**：用户知道信息从哪里来的
2. **可验证**：用户可以点击查看原文
3. **可追溯**：出问题时可以定位错误来源
4. **合规**：某些场景要求标注信息来源

## 方法一：Prompt 引导引用

通过在 System Prompt 中明确要求 LLM 在引用信息的句末标注编号，让生成与引用同步完成。

```python
def generate_with_citation(query: str, docs_with_meta: list, llm) -> str:
    """让 LLM 生成带引用的回答"""
    
    # 构建带编号的上下文
    context_parts = []
    for i, (doc, meta) in enumerate(docs_with_meta):
        source = meta.get('source', meta.get('file', f'来源{i+1}'))
        context_parts.append(f"[{i+1}] (来源: {source})\n{doc}")
    
    context = "\n\n".join(context_parts)
    
    messages = [
        {"role": "system", "content": f"""你是一个知识库问答助手。回答规则：
1. 基于参考资料回答
2. 在引用信息的句末标注来源编号，如[1]、[2]
3. 如果引用了多个来源，标注多个编号，如[1][2]
4. 没有资料来源的内容不要编造
5. 格式：[编号](来源名称)"""},
        {"role": "user", "content": f"参考资料：\n{context}\n\n问题：{query}"}
    ]
    
    response = llm.chat.completions.create(
        model="deepseek-v4-flash",
        messages=messages,
    )
    return response.choices[0].message.content
```

## 方法二：结构化引用输出

利用 JSON Mode 输出包含 answer 字段和独立 citations 数组的结构化结果，便于下游程序解析和展示。

```python
def structured_citation(query: str, docs_with_meta: list) -> dict:
    """输出结构化的带引用回答"""
    
    context_parts = []
    for i, (doc, meta) in enumerate(docs_with_meta):
        source = meta.get('file', meta.get('source', f'来源{i+1}'))
        context_parts.append(f"[{i+1}] {doc}")
    
    context = "\n\n".join(context_parts)
    
    response = client.chat.completions.create(
        model="deepseek-v4-flash",
        messages=[
            {"role": "system", "content": "输出 JSON 格式的回答，包含 answer 和 citations 字段。"},
            {"role": "user", "content": f"""参考资料：
{context}

问题：{query}

输出 JSON：
{{
    "answer": "你的回答",
    "citations": [
        {{"number": 1, "text": "引用的具体句子", "source": "来源文件名"}},
        ...
    ]
}}"""}
        ],
        response_format={"type": "json_object"},
    )
    
    return json.loads(response.choices[0].message.content)

# 使用
docs_with_meta = list(zip(
    collection.query(query_texts=["RAG 是什么？"], n_results=2)['documents'][0],
    collection.query(query_texts=["RAG 是什么？"], n_results=2)['metadatas'][0]
))

result = structured_citation("RAG 是什么？", docs_with_meta)
print(result['answer'])
for c in result.get('citations', []):
    print(f"  [{c['number']}] {c['text'][:50]}... → {c['source']}")
```

## 方法三：Post-hoc 引用匹配

先生成回答，再匹配引用来源：

```python
def posthoc_citation(answer: str, docs: list) -> list:
    """在生成回答后，匹配句子和来源"""
    citations = []
    
    for doc in docs:
        resp = client.chat.completions.create(
            model="deepseek-v4-flash",
            messages=[{
                "role": "user",
                "content": f"""判断以下回答中的哪些内容来自这篇资料。

回答：{answer}

资料：{doc}

输出回答中和资料匹配的句子，每句一行。如果不匹配，输出"无"。"""
            }],
        )
        matched = resp.choices[0].message.content
        if matched and matched != "无":
            citations.append({"source": doc[:80], "matched_sentences": matched})
    
    return citations
```

## 引用可视化

将结构化引用信息格式化为用户友好的展示形式，在回答下方列出来源文件及对应摘要片段。

```python
def format_answer_with_citations(rag_result: dict) -> str:
    """格式化带引用的回答"""
    answer = rag_result['answer']
    sources = rag_result.get('sources', [])
    
    lines = [answer, "\n---\n📚 来源:"]
    for i, src in enumerate(sources, 1):
        file_name = src.get('file', src.get('source', '未知'))
        excerpt = src.get('chunk', src)[:100]
        lines.append(f"  [{i}] {file_name}")
        lines.append(f"      {excerpt}...")
    
    return "\n".join(lines)

# 使用
result = rag.ask("RAG 的优势是什么？")
print(format_answer_with_citations(result))
```

## 引用质量评估

借助 LLM 自动评估引用的准确率、覆盖率以及是否存在未引用的幻觉内容，量化回答的可信度。

```python
def evaluate_citation_quality(answer: str, contexts: list) -> dict:
    """评估引用质量"""
    
    prompt = f"""评估以下回答的引用质量。

回答：{answer}

资料：
{"".join([f"[{i+1}] {c}\n" for i, c in enumerate(contexts)])}

评估标准：
1. 引用准确率：引用是否真实来自资料？
2. 引用覆盖率：关键信息是否有引用？
3. 幻觉率：是否有未引用的编造内容？

输出 JSON：
{{
    "citation_accuracy": 0-10,
    "citation_coverage": 0-10,
    "hallucination": true/false,
    "reason": "..." 
}}"""
    
    resp = client.chat.completions.create(
        model="deepseek-v4-flash",
        messages=[{"role": "user", "content": prompt}],
        response_format={"type": "json_object"},
    )
    
    return json.loads(resp.choices[0].message.content)
```
